// Import the library (development mode)
import * as unsea from './src/index.js';

// Generate keypairs
const keys = await unsea.generateRandomPair();
console.log('Keys:', keys);

// Sign a message
const message = "Hello, World!";
const signature = await unsea.signMessage(message, keys.priv);
const isValid = await unsea.verifyMessage(message, signature, keys.pub);
console.log('Valid signature:', isValid);
